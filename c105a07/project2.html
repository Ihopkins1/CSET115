Original Console Application:

const prompt = require("prompt-sync")();

let option = Number(prompt("Please select an option:\n" + 
    "1. Add\n" +
    "2. Subtract\n"+
    "3. Multiply\n"+
    "4. Divide\n"+
    "5. Quit\n"));   

let first_num, second_num;

if(option === 5) {
    console.log("Goodbye!")
} else {
    first_num = Number(prompt("Please enter a number: \n"));
    second_num = Number(prompt("Enter another number: \n"));
}

if(option === 1) {
    console.log(`The sum is ${first_num + second_num}`);
} else if(option === 2) {
    console.log(`The difference is ${first_num - second_num}`);
} else if(option === 3) {
    console.log(`The product is ${first_num * second_num}`);
} else if(option === 4) {
    if (second_num === 0) {
        console.log("Cannot divide by zero.");
    } else {
        console.log(`The quotient is ${first_num / second_num}`);
    }
} else {
    console.log("Invalid option!");
}

Web Calculator Implementation:

document.getElementById('calculateBtn').addEventListener('click', function() {
    const input = document.getElementById('calcInput').value.trim();
    
    if (!input) {
        document.getElementById('result').textContent = 'Please enter a calculation!';
        return;
    }
    
    try {
        let result;
        
        if (input.includes('+')) {
            const parts = input.split('+');
            if (parts.length === 2) {
                const num1 = parseFloat(parts[0].trim());
                const num2 = parseFloat(parts[1].trim());
                result = num1 + num2;
            }
        } else if (input.includes('-')) {
            const parts = input.split('-');
            if (parts.length === 2) {
                const num1 = parseFloat(parts[0].trim());
                const num2 = parseFloat(parts[1].trim());
                result = num1 - num2;
            }
        } else if (input.includes('*')) {
            const parts = input.split('*');
            if (parts.length === 2) {
                const num1 = parseFloat(parts[0].trim());
                const num2 = parseFloat(parts[1].trim());
                result = num1 * num2;
            }
        } else if (input.includes('/')) {
            const parts = input.split('/');
            if (parts.length === 2) {
                const num1 = parseFloat(parts[0].trim());
                const num2 = parseFloat(parts[1].trim());
                if (num2 === 0) {
                    document.getElementById('result').textContent = 'Cannot divide by zero!';
                    return;
                }
                result = num1 / num2;
            }
        }
        
        if (result !== undefined && !isNaN(result)) {
            document.getElementById('result').textContent = `${input} = ${result}`;
        } else {
            document.getElementById('result').textContent = 'Invalid calculation format';
        }
        
    } catch (error) {
        document.getElementById('result').textContent = 'Error in calculation';
    }
});

Project Overview

This project evolved from a simple console-based calculator to an interactive web application. The original version used prompts and console output, while the web version provides a modern user interface with real-time calculations.

The calculator handles edge cases like division by zero and provides clear feedback to users for all operations.
